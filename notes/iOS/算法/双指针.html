# 盛最多水的容器

/**
 给定一个长度为 n 的整数数组 height 。有 n 条垂线，第 i 条线的两个端点是 (i, 0) 和 (i, height[i]) 。
 找出其中的两条线，使得它们与 x 轴共同构成的容器可以容纳最多的水。返回容器可以储存的最大水量

 注意： 你不能倾斜容器， n 至少是2。
 
 输入: [1, 8, 6, 2, 5, 4, 8, 3, 7]
 输出: 49
 */
- (void)containerWithMostWater {
    NSArray *heightArray = @[@(1),@(8),@(6),@(2),@(5),@(4),@(8),@(3),@(7)];
    int startX = 0;
    int endX = (int)(heightArray.count - 1);
    
    int maxArea = 0;
    while (startX < endX) {
        // 计算宽度
        int width = endX - startX;
        // 取高度最小的
        int leftHeight = [heightArray[startX] intValue];
        int rightHeight = [heightArray[endX] intValue];
        int height = MIN(leftHeight, rightHeight);
        
        if (width * height > maxArea) {
            maxArea = width * height;
        }
        // 肯定需要高度小的垂直线移动，因为宽度-1变小，高度高的移动的话，变得再高也没用，水位由最低水位决定，所以需要移动高度小的垂直线
        // 左边高度大，右侧坐标左移；反之左侧坐标右移
        leftHeight > rightHeight ? endX -- : startX++;
    }
    NSLog(@"maxArea--->%d", maxArea);
}

# 接雨水

/**
 给定 n 个非负整数表示每个宽度为 1 的柱子的高度图，计算按此排列的柱子，下雨之后能接多少雨水。
 输入：height = [0,1,0,2,1,0,1,3,2,1,2,1]
 输出：6
 解释：上面是由数组 [0,1,0,2,1,0,1,3,2,1,2,1] 表示的高度图，在这种情况下，可以接 6 个单位的雨水（蓝色部分表示雨水）。
 图片地址：https://assets.leetcode-cn.com/aliyun-lc-upload/uploads/2018/10/22/rainwatertrap.png
 
 双指针解法：https://leetcode-cn.com/problems/trapping-rain-water/solution/jie-yu-shui-by-leetcode-solution-tuvc/
 */
- (void)receiveRainwater {
//    NSArray *heightArray = @[@(0), @(1), @(0), @(2), @(1), @(0), @(1), @(3), @(2), @(1), @(2), @(1)];
    NSArray *heightArray = @[@(4), @(2), @(0), @(3), @(2), @(5)];
    
    int rainwaterSum = 0;
    int left = 0, right = (int)(heightArray.count) - 1;
    int leftMax = 0, rightMax = 0;
    while (left < right) {
        int leftValue = [heightArray[left] intValue];
        int rightValue = [heightArray[right] intValue];
        leftMax = MAX(leftMax, leftValue);
        rightMax = MAX(rightMax, rightValue);
        if (leftValue < rightValue) {
            rainwaterSum += leftMax - leftValue;
            ++left;
        } else {
            rainwaterSum += rightMax - rightValue;
            --right;
        }
    }
    NSLog(@"rainwaterSum--->%d", rainwaterSum);
}
