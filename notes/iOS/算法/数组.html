# 两个有序数组合并成一个有序数组

- (void)combineTwoOrderedArray2OneOrderedArray {
    NSArray *orderedArray1 = @[@(1), @(2), @(5)];
    NSArray *orderedArray2 = @[@(3), @(7), @(9)];
    NSMutableArray *newArray = [self compareWithArray:orderedArray1 secondArray:orderedArray2];
    NSLog(@"newArray--->%@", newArray);
}

- (NSMutableArray *)compareWithArray:(NSArray *)firstArray secondArray:(NSArray *)secondArray {
    NSMutableArray *newArray = [NSMutableArray array];
    int firstIndex = 0, secondIndex = 0;// 记录每个数组遍历的索引
    for (int i = 0; i < firstArray.count; i ++) {
        int value1 = [firstArray[i] intValue];
        for (int j = secondIndex; j < secondArray.count; j ++) {// j = secondIndex，避免前面重复比较
            int value2 = [secondArray[j] intValue];
            if (value1 > value2) {
                secondIndex = j + 1;// 比较完将较小值游标往右偏移
                [newArray addObject:@(value2)];
            } else {
                firstIndex = i + 1;// 比较完将较小值游标往右偏移
                [newArray addObject:@(value1)];
                break;
            }
        }
    }
    // 将未比较完的数组中的值加入到新数组末尾
    if (firstIndex == firstArray.count && secondIndex != secondArray.count) {
        for (int i = secondIndex; i < secondArray.count; i ++) {
            [newArray addObject:@([secondArray[i] intValue])];
        }
    } else if (firstIndex != firstArray.count && secondIndex == secondArray.count) {
        for (int i = firstIndex; i < firstArray.count; i ++) {
            [newArray addObject:@([firstArray[i] intValue])];
        }
    }
    return newArray;
}

# 找出指定数组中等于指定值的三个元素

/**
 给定一个包含 n 个整数的数组 nums ，判断 nums 中是否存在三个元素 a，b，c ， 使得 a + b + c = 0 ？找出所有满足条件且不重复的三元组。
 注意： 答案中不可以包含重复的三元组。

 例如, 给定数组 nums = [-1, 0, 1, 2, -1, -4]，
 满足要求的三元组集合为：
 [
 [-1, 0, 1],
 [-1, -1, 2]
 ]
 */
- (void)findThreeNumInArrayEqualTarget {
    NSArray *numArray = @[@(-1), @(0), @(1), @(2), @(-1), @(-4)];
    int sum = 0;
    
    NSMutableArray *resultArray = [NSMutableArray array];
    for (int i = 0; i < numArray.count; i ++) {
        for (int j = i + 1; j < numArray.count; j ++) {
            for (int k = j + 1; k < numArray.count; k ++) {
                if ([numArray[i] intValue] + [numArray[j] intValue] + [numArray[k] intValue] == sum) {
                    BOOL isContainSameValue = NO;
                    // 遍历结果数组，不要把重复的三元组加入结果数组中
                    for (int l = 0; l < resultArray.count; l ++) {
                        if (numArray[i] == resultArray[l][1] && numArray[j] == resultArray[l][2]) {
                            isContainSameValue = YES;
                        }
                    }
                    if (!isContainSameValue) {
                        [resultArray addObject:@[numArray[i], numArray[j], numArray[k]]];
                    }
                }
            }
        }
    }
    NSLog(@"resultArray--->%@", resultArray);
}

# 下一个更大数的排列

/**
 实现获取下一个排列的函数，算法需要将给定数字序列重新排列成字典序中下一个更大的排列。
 如果不存在下一个更大的排列，则将数字重新排列成最小的排列（即升序排列）。
 必须 原地 修改，只允许使用额外常数空间。
 以下是一些例子，输入位于左侧列，其相应输出位于右侧列。
 1,2,3 → 1,3,2
 3,2,1 → 1,2,3
 1,5,8,6,7,3,2,1 --> 1,5,8,7,1,2,3,6
 1,5,8,6,9,7,2,1 --> 1,5,8,7,1,2,6,9
 */
- (void)nextPermutation {
    /**
     1.首先找到从右往左的第一对nums[i-1] < nums[i]的一对数
     2.然后在nums[i-1]的右边的序列中按照从右往左的顺序找到第一个大于nums[i-1]的元素nums[j]，将nums[i-1]与nums[j]交换
     3.对nums[i-1]右边的序列进行翻转，使其变成升序子序列
     */
    NSMutableArray *numArray = [@[@(1), @(5), @(8), @(6), @(9), @(7), @(2), @(1)] mutableCopy];
    // 找出nums[i-1]
    int firstSwapIndex = 0;
    for (int i = (int)(numArray.count - 1); i > 0; i --) {
        if ([numArray[i - 1] intValue] < [numArray[i] intValue]) {
            firstSwapIndex = i - 1;
            break;
        }
    }
    // 找出nums[j]
    int secondSwapIndex = 0;
    for (int i = (int)(numArray.count - 1); i > firstSwapIndex; i --) {
        if ([numArray[i] intValue] > [numArray[firstSwapIndex] intValue]) {
            secondSwapIndex = i;
            break;
        }
    }
    // 交换
    id firstValue = numArray[firstSwapIndex];
    numArray[firstSwapIndex] = numArray[secondSwapIndex];
    numArray[secondSwapIndex] = firstValue;
    // 拆分出前后两部分数组
    NSMutableArray *leftArray = [NSMutableArray array];
    NSMutableArray *rightArray = [NSMutableArray array];
    for (int i = 0; i < numArray.count; i ++) {
        if (i <= firstSwapIndex) {
            [leftArray addObject:numArray[i]];
        } else {
            [rightArray addObject:numArray[i]];
        }
    }
    // 将firstSwapIndex后面的数字按升序排列
    for (int i = 0; i < rightArray.count - 1; i ++) {
        for (int j = i + 1; j < rightArray.count; j ++) {// j = i + 1，直接从下一位开始遍历
            if ([rightArray[i] intValue] > [rightArray[j] intValue]) {
                int temp = [rightArray[j] intValue];
                rightArray[j] = @([rightArray[i] intValue]);
                rightArray[i] = @(temp);
            }
        }
    }
    NSMutableArray *resultArray = [NSMutableArray arrayWithArray:leftArray];
    [resultArray addObjectsFromArray:rightArray];
    NSLog(@"resultArray--->%@", resultArray);
}

# 搜索旋转排序数组

/**
 假设按照升序排序的数组在预先未知的某个点上进行了旋转。
 ( 例如，数组 [0,1,2,4,5,6,7] 可能变为 [4,5,6,7,0,1,2] )。
 搜索一个给定的目标值，如果数组中存在这个目标值，则返回它的索引，否则返回 -1 。
 你可以假设数组中不存在重复的元素。
 你的算法时间复杂度必须是 O(log n) 级别。
 示例：
 输入: nums = [4,5,6,7,0,1,2], target = 0
 输出: 4

 输入: nums = [4,5,6,7,0,1,2], target = 3
 输出: -1
 */
- (void)searchInRotatedSortedArray {
    NSArray *numArray = @[@(4), @(5), @(6), @(7), @(0), @(1), @(2)];
    int target = 0;
    
    int left = 0;
    int right = (int)(numArray.count - 1);
    int result = -1;
    while (left <= right) {
        int mid = left + (right - left)/2;// 中位数索引
        if ([numArray[mid] intValue] == target) {
            result = mid;
            break;// 找到匹配值，退出循环
        } else if ([numArray[mid] intValue] >= [numArray[left] intValue]) {// 中位数>=左边的，说明左边的是升序的
            // 判断target是否在左侧序列中，如果在，right游标移到mid-1处，在左侧进行遍历；反之left游标移到mid+1处，在右侧进行遍历
            if ([numArray[left] intValue] <= target && target < [numArray[mid] intValue]) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        } else if ([numArray[mid] intValue] < [numArray[left] intValue]) {// 中位数<左边的，说明右边的是升序的
            if ([numArray[mid] intValue] < target && target <= [numArray[right] intValue]) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
    }
    NSLog(@"result--->%d", result);
}

# 总和等于目标值的所有可能组合

/**
 给定一个 无重复元素 的数组 candidates 和一个目标数 target ，找出 candidates 中所有可以使数字和为 target 的组合。
 candidates 中的数字可以无限制重复被选取。
 说明：
 所有数字（包括 target ）都是正整数。
 解集不能包含重复的组合。
 示例 1:
 输入: candidates = [2,3,6,7], target = 7,
 所求解集为:
 [
   [7],
   [2,2,3]
 ]

 示例 2:
 输入: candidates = [2,3,5], target = 8,
 所求解集为:
 [
   [2,2,2,2],
   [2,3,3],
   [3,5]
 ]
 */
- (void)combinationSum {
    NSArray *array = @[@(2), @(3), @(5)];
    int target = 8;
    
    NSMutableArray<NSMutableArray *> *resultArray = [NSMutableArray array];
    [self array:array target:target resultArray:resultArray start:0 tempArray:[NSMutableArray array]];
    NSLog(@"resultArray--->%@", resultArray);
}

- (void)array:(NSArray *)array target:(int)target resultArray:(NSMutableArray<NSMutableArray *> *)resultArray start:(int)start tempArray:(NSMutableArray *)tempArray {
    if (target == 0) {
        [resultArray addObject:[NSMutableArray arrayWithArray:tempArray]];// 关键点2：要新创建一个数组
        return;
    } else if (target < 0) {
        return;
    }
    for (int i = start; i < array.count; i ++) {// 关键点3：遍历从i开始，从当前元素开始遍历。如果是0，还会导致死循环
        int value = [array[i] intValue];
        [tempArray addObject:array[i]];
        [self array:array target:target-value resultArray:resultArray start:i tempArray:tempArray];
        // 关键点1：如果target-value<0，将加入的元素移除；如果满足条件target-value=0，会新创建一个数组对象，不影响结果
        [tempArray removeLastObject];
    }
}

# 最大子数组和

/**
 给你一个整数数组 nums ，请你找出一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。
 子数组 是数组中的一个连续部分。
 */
- (void)maxSumOfSubArray {
    NSArray *array = @[@(-2), @(1), @(-3), @(4), @(-1), @(2), @(1), @(-5), @(4)];
    int maxSum = [array[0] intValue];
    for (NSInteger i = 0; i < array.count; i ++) {
        int sum = [array[i] intValue];
        for (NSInteger j = i + 1; j < array.count; j ++) {
            sum += [array[j] intValue];
            if (sum > maxSum) {
                maxSum = sum;
            }
        }
    }
    NSLog(@"maxSum--->%d", maxSum);
}

# 缺失的第一个正数

/**
 给你一个未排序的整数数组 nums ，请你找出其中没有出现的最小的正整数。
 实现时间复杂度为 O(n) 并且只使用常数级别额外空间的解决方案。
 示例 1：
 输入：nums = [1,2,0]
 输出：3
 
 示例 2：
 输入：nums = [3,4,-1,1]
 输出：2
 
 示例 3：
 输入：nums = [7,8,9,11,12]
 输出：1
 
 我们可以将数组所有的数放入哈希表，随后从1开始依次枚举正整数，并判断其是否在哈希表中；时间复杂度O(N)，空间复杂度O(N)
 我们可以从1开始依次枚举正整数，并遍历数组，判断其是否在数组中。时间复杂度O(N²)，空间复杂度O(1)
 */
- (void)firstMissingPositive {
    NSMutableArray *numArray = [@[@(3), @(4), @(-1), @(1)] mutableCopy];
    int count = (int)numArray.count;
    // 置换归位，如3应该出现在索引2的位置，即value-1，3要与numArray[value-1]值进行交换
    for (int i = 0; i < count; i ++) {
//        int value = [numArray[i] intValue];
//        int rightPositionValue = [numArray[value-1] intValue];
        // 如果值本身就在正确的位置上，会陷入死循环，要跳过value == rightPositionValue的情况
        while ([numArray[i] intValue] > 0 && [numArray[i] intValue] <= count && [numArray[i] intValue] != [numArray[[numArray[i] intValue]-1] intValue]) {
            id temp = numArray[i];
            numArray[i] = numArray[[numArray[i] intValue]-1];
            numArray[[temp intValue]-1] = temp;
        }
    }
    int missingPositive = 0;
    for (int i = 0; i < count; i ++) {
        if ([numArray[i] intValue] != i + 1) {
            missingPositive = i + 1;
        }
    }
    if (missingPositive == 0) {
        missingPositive = count + 1;
    }
    NSLog(@"missingPositive--->%d", missingPositive);
}

# 合并区间

/**
 以数组 intervals 表示若干个区间的集合，其中单个区间为 intervals[i] = [starti, endi] 。请你合并所有重叠的区间，并返回 一个不重叠的区间数组，该数组需恰好覆盖输入中的所有区间 。
 
 示例 1：
 输入：intervals = [[1,3],[2,6],[8,10],[15,18]]
 输出：[[1,6],[8,10],[15,18]]
 解释：区间 [1,3] 和 [2,6] 重叠, 将它们合并为 [1,6].
 */
- (void)intervalMerge {
    NSArray *intervalArray = @[@[@(1),@(3)], @[@(2),@(6)], @[@(15),@(18)], @[@(8),@(10)]];
    // 1.先根据左区间值排个序
    NSMutableArray *sortArray = [NSMutableArray array];
    for (int i = 0; i < intervalArray.count; i ++) {
        if (i == 0) {
            [sortArray addObject:intervalArray[i]];
        } else {
            NSArray *array = intervalArray[i];// 要比较的数组
            int index = 0;
            for (int j = 0; j < sortArray.count; j ++) {
                NSArray *orderArray = intervalArray[j];// 已经排序好的数组
                if ([array[0] intValue] < [orderArray[0] intValue]) {
                    break;
                } else {
                    index ++;
                }
            }
            [sortArray insertObject:array atIndex:index];
        }
    }
    // 2.排序完成后，第n个数组A和第n+1个数组B只有3种可能，A包含B，A和B相交，A和B不相交
    NSMutableArray *resultArray = [NSMutableArray array];
    for (int i = 0; i < sortArray.count; i ++) {
        if (i == 0) {
            [resultArray addObject:[sortArray[i] mutableCopy]];
        } else {
            NSArray *array = sortArray[i];// 要比较的数组
            NSMutableArray *orderArray = [resultArray lastObject];// 已经排序好的数组的最后一个元素即可
            if ([array[0] intValue] > [orderArray[1] intValue]) {// A和B不相交，B左区间值>A右区间值，新区间
                [resultArray addObject:array];
            } else if ([array[1] intValue] <= [orderArray[1] intValue]) {// A包含B，跳过B，继续遍历下一个数组
                continue;
            } else if ([array[1] intValue] > [orderArray[1] intValue]) {// A和B相交，即[A的左区间值,B的右区间值]
                orderArray[1] = array[1];
            }
        }
    }
    NSLog(@"resultArray--->%@", resultArray);
}
