# 有效的数独

/**
 判断一个 9x9 的数独是否有效。只需要 根据以下规则 ，验证已经填入的数字是否有效即可。

 数字 1-9 在每一行只能出现一次。
 数字 1-9 在每一列只能出现一次。
 数字 1-9 在每一个以粗实线分隔的 3x3 宫内只能出现一次。
 
 数组部分空格内已填入数字，空白格用'.'表示
 */

- (void)validSudoku {
    NSArray *sudokuArray = @[@[@"5", @"3", @".", @".", @"7", @".", @".", @".", @"."],
                             @[@"6", @".", @".", @"1", @"9", @"5", @".", @".", @"."],
                             @[@".", @"9", @"8", @".", @".", @".", @".", @"6", @"."],
                             @[@"8", @".", @".", @".", @"6", @".", @".", @".", @"3"],
                             @[@"4", @".", @".", @"8", @".", @"3", @".", @".", @"1"],
                             @[@"7", @".", @".", @".", @"2", @".", @".", @".", @"6"],
                             @[@".", @"6", @".", @".", @".", @".", @"2", @"8", @"."],
                             @[@".", @".", @".", @"4", @"1", @"9", @".", @".", @"5"],
                             @[@".", @".", @".", @".", @"8", @".", @".", @"7", @"9"]];
    BOOL isValidSudoku = [self isValidSudoku:sudokuArray];
    NSLog(@"isValidSudoku--->%@", isValidSudoku ? @"YES" : @"NO");
    
}

- (BOOL)isValidSudoku:(NSArray *)sudokuArray {
    for (int i = 0; i < 9; i ++) {
        // 横排数组
        NSArray *rowArray = sudokuArray[i];
        // 纵排数组
        NSArray *columnArray = @[sudokuArray[0][i], sudokuArray[1][i], sudokuArray[2][i],
                                 sudokuArray[3][i], sudokuArray[4][i], sudokuArray[5][i],
                                 sudokuArray[6][i], sudokuArray[7][i], sudokuArray[8][i]];
        if ([self isContainSameNum:rowArray] || [self isContainSameNum:columnArray]) {
            return NO;
        }
    }
//    NSMutableArray *squareArray = [NSMutableArray array];
    for (int k = 0; k < 9; k += 3) {// 2.最外层再嵌套一个for循环往右移动
        for (int i = 0; i < 9; i += 3) {// 1.取出最左边的3个九宫格
            NSMutableArray *array = [NSMutableArray array];
            for (int j = 0; j < 3; j ++) {
                [array addObject:sudokuArray[i][j+k]];
                [array addObject:sudokuArray[i+1][j+k]];
                [array addObject:sudokuArray[i+2][j+k]];
            }
            if ([self isContainSameNum:array]) {
                return NO;
            }
//            [squareArray addObject:array];
        }
    }
    
    return YES;
}

/** 数组中是否包含相同的数字 */
- (BOOL)isContainSameNum:(NSArray *)array {
    for (int i = 0; i < array.count - 2; i ++) {
        if (![array[i] isEqualToString:@"."]) {// 不等于.的元素才去比较
            for (int j = i + 1; j < array.count; j ++) {
                if ([array[i] isEqualToString:array[j]]) {
                    return YES;
                }
            }
        }
    }
    return NO;
}
