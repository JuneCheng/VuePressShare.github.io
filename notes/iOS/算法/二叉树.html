# 二叉树翻转

- (void)binaryTreeReverse {
    BinaryTreeNode *node = [self createBinaryTree];
    
    [self mirrorOfBinaryTree:node];
    NSLog(@"二叉树翻转--->%@", node);
}

- (BinaryTreeNode *)mirrorOfBinaryTree:(BinaryTreeNode *)root{
    BinaryTreeNode *temp;
    if (root) {
        [self mirrorOfBinaryTree:root.leftNode];
        [self mirrorOfBinaryTree:root.rightNode];
        temp = root.leftNode;
        root.leftNode = root.rightNode;
        root.rightNode = temp;
    }
    return root;
}

# 从根节点到叶子节点路径总和等于目标值的路径

/**
 给你二叉树的根节点 root 和一个整数目标和 targetSum ，找出所有 从根节点到叶子节点 路径总和等于给定目标和的路径。
 此题与另一题比较类似：总和等于目标值的所有可能组合
 */
- (void)pathOfRoot2Leaf {
    BinaryTreeNode *node = [self createBinaryTree];
    int target = 10;
    
    NSMutableArray *resultArray = [NSMutableArray array];
    [self node:node target:target resultArray:resultArray tempArray:[NSMutableArray array]];
    NSLog(@"resultArray--->%@", resultArray);
}

- (void)node:(BinaryTreeNode *)node target:(int)target resultArray:(NSMutableArray *)resultArray tempArray:(NSMutableArray *)tempArray {
    if (target < 0) {
        return;
    }
    if (!node) {
        return;
    }
    int value = (int)(node.value);
    [tempArray addObject:@(value)];
    if (target - value == 0) {// 关键点2：如果等于0，要新创建一个数组
        [resultArray addObject:[NSMutableArray arrayWithArray:tempArray]];// 关键点2：要新创建一个数组
    } else {
        [self node:node.leftNode target:target-value resultArray:resultArray tempArray:tempArray];
        [self node:node.rightNode target:target-value resultArray:resultArray tempArray:tempArray];
        // 关键点1：如果target-value<0，将加入的元素移除；
        [tempArray removeLastObject];
    }
}

# 二叉树的层次遍历

/**
 给你二叉树的根节点 root ，返回其节点值的 层序遍历 。 （即逐层地，从左到右访问所有节点）。
 示例1 :
 输入：root = [3,9,20,null,null,15,7]
 输出：[[3],[9,20],[15,7]]
 
 示例2 :
 输入：root = [1,2,3,4,5,6,7]
 输出：[[1], [2,3], [4,5,6,7]]
 */
- (void)levelOrder {
    BinaryTreeNode *node = [self createBinaryTree];
    
    NSMutableArray *resultArray = [NSMutableArray array];
    [self node:node level:0 resultArray:resultArray];
    NSLog(@"resultArray--->%@", resultArray);
}

- (void)node:(BinaryTreeNode *)node level:(int)level resultArray:(NSMutableArray *)resultArray {
    // 拦截
    if (!node) {
        return;
    }
    // 如果层级 >= 数组的长度，创建新一层的新数组
    if (level >= resultArray.count) {
        NSMutableArray *nexSubArray = [NSMutableArray array];
        [resultArray addObject:nexSubArray];
    }
    [resultArray[level] addObject:@(node.value)];// 加入该元素
    
    // 下一层，继续递归
    level ++;
    [self node:node.leftNode level:level resultArray:resultArray];
    [self node:node.rightNode level:level resultArray:resultArray];
}

# 二叉树的右视图

/**
 给定一个二叉树的 根节点 root，想象自己站在它的右侧，按照从顶部到底部的顺序，返回从右侧所能看到的节点值。
 示例 1:
 输入: [1,2,3,null,5,null,4]
 输出: [1,3,4]
 
 示例 2:
 输入: [1,null,3]
 输出: [1,3]
 
 此题及在题目"二叉树的层次遍历"的基础上，遍历数组，取出最后一个节点
 */
- (void)rightSideView {
    BinaryTreeNode *node = [self createBinaryTree];
    
    NSMutableArray *resultArray = [NSMutableArray array];
    [self node:node level:0 resultArray:resultArray];
    
    NSMutableArray *array = [NSMutableArray array];
    for (int i = 0; i < resultArray.count; i ++) {
        [array addObject:[resultArray[i] lastObject]];
    }
    NSLog(@"array--->%@", array);
}

# 二叉树的锯齿层次遍历

/**
 给你二叉树的根节点 root ，返回其节点值的 锯齿形层序遍历 。（即先从左往右，再从右往左进行下一层遍历，以此类推，层与层之间交替进行）。
 示例1 :
 输入：root = [3,9,20,null,null,15,7]
 输出：[[3],[20,9],[15,7]]
 
 示例2 :
 输入：root = [1,2,3,4,5,6,7]
 输出：[[1], [3,2], [4,5,6,7]]
 */
- (void)zigzagLevelOrder {
    BinaryTreeNode *node = [self createBinaryTree];
    
    NSMutableArray *resultArray = [NSMutableArray array];
    [self node:node level:0 forward:YES resultArray:resultArray];
    NSLog(@"resultArray--->%@", resultArray);
}

- (void)node:(BinaryTreeNode *)node level:(int)level forward:(BOOL)isForward resultArray:(NSMutableArray *)resultArray {
    // 拦截
    if (!node) {
        return;
    }
    // 如果层级 >= 数组的长度，创建新一层的新数组
    if (level >= resultArray.count) {
        NSMutableArray *nexSubArray = [NSMutableArray array];
        [resultArray addObject:nexSubArray];
    }
    if (isForward) {
        [resultArray[level] addObject:@(node.value)];// 加入该元素
    } else {
        [resultArray[level] insertObject:@(node.value) atIndex:0];// 逆序加入该元素
    }
    
    // 下一层，继续递归
    level ++;
    isForward = !isForward;
    [self node:node.leftNode level:level forward:isForward resultArray:resultArray];
    [self node:node.rightNode level:level forward:isForward resultArray:resultArray];
}

# 二叉树的最近公共祖先

/**
 给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。
 */
- (void)lowestCommonAncestor {
    BinaryTreeNode *node1 = [BinaryTreeNode initWithValue:1];
    BinaryTreeNode *node2 = [BinaryTreeNode initWithValue:2];
    BinaryTreeNode *node3 = [BinaryTreeNode initWithValue:3];
    BinaryTreeNode *node4 = [BinaryTreeNode initWithValue:4];
    BinaryTreeNode *node5 = [BinaryTreeNode initWithValue:5];
    BinaryTreeNode *node6 = [BinaryTreeNode initWithValue:6];
    BinaryTreeNode *node7 = [BinaryTreeNode initWithValue:7];
    node1.leftNode = node2;
    node1.rightNode = node3;
    node2.leftNode = node4;
    node2.rightNode = node5;
    node3.leftNode = node6;
    node3.rightNode = node7;
    
    BinaryTreeNode *resultNode = [self lowestCommonAncestorWithRoot:node1 pNode:node4 qNode:node6];
    NSLog(@"resultNode--->%@", resultNode);
}

- (BinaryTreeNode *)lowestCommonAncestorWithRoot:(BinaryTreeNode *)root pNode:(BinaryTreeNode *)pNode qNode:(BinaryTreeNode *)qNode {
    if (root == pNode || root == qNode) {
        return root;
    }
    if (root) {
        BinaryTreeNode *lNode = [self lowestCommonAncestorWithRoot:root.leftNode pNode:pNode qNode:qNode];
        BinaryTreeNode *rNode = [self lowestCommonAncestorWithRoot:root.rightNode pNode:pNode qNode:qNode];
        if (lNode && rNode) {
            return root;
        } else if (!lNode) {// 两个都在右子树
            return rNode;
        } else {// 两个都在左子树
            return lNode;
        }
    }
    return nil;
}

# 二叉树的完整性检验

/**
 给定一个二叉树的 root ，确定它是否是一个完全二叉树 。
 在一个完全二叉树中，除了最后一个关卡外，所有关卡都是完全被填满的，并且最后一个关卡中的所有节点都是尽可能靠左的。它可以包含 1 到 2h 节点之间的最后一级 h 。
 
 我们可以找到规律：对于一个节点，它的左孩子为：(depth, position) -> (depth + 1, position * 2)，右孩子为 (depth, position) -> (depth + 1, position * 2 + 1)。所以，对于深度为 d 的行恰好含有
 2 ^ (d−1) 个节点，所有节点都是靠左边排列的当他们的位置编号是 0, 1, ... 且没有间隙。
 */
- (void)isCompleteTree {
    BinaryTreeNode *node = [self createBinaryTree];

    NSMutableArray<AssistNode *> *nodeArray = [NSMutableArray array];
    [nodeArray addObject:[AssistNode initWithCode:1 node:node]];
    int i = 0;
    while (i < nodeArray.count) {
        AssistNode *assistNode = nodeArray[i++];// 这里先取i值，再进行i++的
        if (assistNode.node) {
            [nodeArray addObject:[AssistNode initWithCode:assistNode.code*2 node:assistNode.node.leftNode]];
            [nodeArray addObject:[AssistNode initWithCode:assistNode.code*2+1 node:assistNode.node.rightNode]];
        }
    }
    
    NSLog(@"isCompleteTree--->%@", nodeArray[i-1].code == nodeArray.count ? @"YES" : @"NO");
}

# 二叉树对称

/** 给你一个二叉树的根节点 root ， 检查它是否轴对称。
 输入：root = [1,2,2,3,4,4,3]
 输出：true
 */
- (void)isSymmetric {
    BinaryTreeNode *root = [self createBinaryTree];
    BOOL isSymmetric = [self checkWithPNode:root.leftNode qNode:root.rightNode];
    NSLog(@"isSymmetric--->%@", isSymmetric ? @"YES" : @"NO");
}

- (BOOL)checkWithPNode:(BinaryTreeNode *)pNode qNode:(BinaryTreeNode *)qNode {
    if (!pNode && !qNode) {
        return YES;
    } else if (!pNode || !qNode) {
        return NO;
    }
    return pNode.value == qNode.value &&
        [self checkWithPNode:pNode.leftNode qNode:qNode.rightNode] &&
        [self checkWithPNode:pNode.rightNode qNode:qNode.leftNode];
}

# 从前序与中序遍历序列构造二叉树

/**
 给定两个整数数组 preorder 和 inorder ，其中 preorder 是二叉树的先序遍历， inorder 是同一棵树的中序遍历，请构造二叉树并返回其根节点。
 
 前序遍历的形式总是 [ 根节点, [左子树的前序遍历结果], [右子树的前序遍历结果] ]，即根节点总是前序遍历中的第一个节点
 中序遍历的形式总是 [ [左子树的中序遍历结果], 根节点, [右子树的中序遍历结果] ]
 只要我们在中序遍历中定位到根节点，那么我们就可以分别知道左子树和右子树中的节点数目。由于同一颗子树的前序遍历和中序遍历的长度显然是相同的，因此我们就可以对应到前序遍历的结果中，对上述形式中的所有左右括号进行定位。

 这样以来，我们就知道了左子树的前序遍历和中序遍历结果，以及右子树的前序遍历和中序遍历结果，我们就可以递归地对构造出左子树和右子树，再将这两颗子树接到根节点的左右位置。
 */
- (void)buildTree {
    NSArray *preOrder = @[@(3), @(9), @(20), @(15), @(7)];
    NSArray *inOrder = @[@(9), @(3), @(15), @(20), @(7)];
    
    NSMutableDictionary *inOrderDic = [NSMutableDictionary dictionary];
    int count = (int)preOrder.count;
    for (int i = 0; i < count; i ++) {
        inOrderDic[inOrder[i]] = @(i);// 索引作为value，值作为key
    }
    BinaryTreeNode *root = [self buildTreeWithPreOrder:preOrder inOrderDic:[inOrderDic copy] preLeft:0 preRight:count-1 inLeft:0 inRight:count-1];
    NSLog(@"root--->%@", root);
}

- (BinaryTreeNode *)buildTreeWithPreOrder:(NSArray *)preOrder inOrderDic:(NSDictionary *)inOrderDic preLeft:(int)preLeft preRight:(int)preRight inLeft:(int)inLeft inRight:(int)inRight {
    if (preLeft > preRight || inLeft > inRight) {
        return nil;
    }
    // 前序遍历中的第一个节点就是根节点
    int index_preOrder = preLeft;
    id value = preOrder[index_preOrder];
    // 在中序遍历中定位根节点
    int index_inOrder = [inOrderDic[value] intValue];
    // 先把根节点建立出来
    BinaryTreeNode *root = [BinaryTreeNode initWithValue:[value intValue]];
    // 得到左子树中的节点数目
    int size_left_subTree = index_inOrder - inLeft;
    // 递归地构造左子树，并连接到根节点
    // 先序遍历中「从 左边界+1 开始的 size_left_subtree」个元素就对应了中序遍历中「从 左边界 开始到 根节点定位-1」的元素
    root.leftNode = [self buildTreeWithPreOrder:preOrder inOrderDic:inOrderDic preLeft:preLeft+1 preRight:preLeft+size_left_subTree inLeft:inLeft inRight:index_inOrder-1];
    // 递归地构造右子树，并连接到根节点
    // 先序遍历中「从 左边界+1+左子树节点数目 开始到 右边界」的元素就对应了中序遍历中「从 根节点定位+1 到 右边界」的元素
    root.rightNode = [self buildTreeWithPreOrder:preOrder inOrderDic:inOrderDic preLeft:preLeft+size_left_subTree+1 preRight:preRight inLeft:index_inOrder+1 inRight:inRight];
    
    return root;
}

/** 给定一个二叉树的根节点 root ，返回 它的 中序 遍历 。 */
- (void)inorderTraversal {
    BinaryTreeNode *root = [self createBinaryTree];
    NSMutableArray *resultArray = [NSMutableArray array];
    [self inorderTraversal:root resultArray:resultArray];
    NSLog(@"resultArray--->%@", resultArray);
}

- (void)inorderTraversal:(BinaryTreeNode *)node resultArray:(NSMutableArray *)resultArray {
    if (!node) {
        return;
    }
    [self inorderTraversal:node.leftNode resultArray:resultArray];
    [resultArray addObject:@(node.value)];
    [self inorderTraversal:node.rightNode resultArray:resultArray];
}

# 二叉树的直径

/**
 给定一棵二叉树，你需要计算它的直径长度。一棵二叉树的直径长度是任意两个结点路径长度中的最大值。这条路径可能穿过也可能不穿过根结点。
 */
- (void)diameterOfBinaryTree {
    BinaryTreeNode *root = [self createBinaryTree];
    // 辅助类，code记录长度
    AssistNode *assistNode = [[AssistNode alloc] init];
    assistNode.code = 1;
    [self depth:root assistNode:assistNode];
    NSLog(@"二叉树的直径--->%ld", assistNode.code-1);
}

- (int)depth:(BinaryTreeNode *)node assistNode:(AssistNode *)assistNode {
    if (!node) {// 访问到空节点了，返回0
        return 0;
    }
    int L = [self depth:node.leftNode assistNode:assistNode];
    int R = [self depth:node.rightNode assistNode:assistNode];
    
    assistNode.code = MAX(assistNode.code, L+R+1);
    
    return MAX(L, R) + 1; // 返回该节点为根的子树的深度
}

# 二叉树的最大深度

/**
 给定一个二叉树，找出其最大深度。
 二叉树的深度为根节点到最远叶子节点的最长路径上的节点数。
 说明: 叶子节点是指没有子节点的节点。
 */
- (void)maxDepth {
    BinaryTreeNode *root = [self createBinaryTree];
    int maxDepth = [self maxDepth:root];
    NSLog(@"maxDepth--->%d", maxDepth);
}

- (int)maxDepth:(BinaryTreeNode *)root {
    if (!root) {
        return 0;
    } else {
        int leftHeight = [self maxDepth:root.leftNode];
        int rightHeight = [self maxDepth:root.rightNode];
        return MAX(leftHeight, rightHeight) + 1;
    }
}
