# 两个有序链表合并成一个有序链表

- (void)combineTwoOrderedLinkedLists2OneOrderedLinkedList {
    // 125  379
    NodeModel *node1 = [NodeModel initWithValue:1];
    NodeModel *node2 = [NodeModel initWithValue:2];
    NodeModel *node3 = [NodeModel initWithValue:5];
    NodeModel *node4 = [NodeModel initWithValue:3];
    NodeModel *node5 = [NodeModel initWithValue:7];
    NodeModel *node6 = [NodeModel initWithValue:9];
    
    node1.next = node2;
    node2.next = node3;
    node4.next = node5;
    node5.next = node6;
    
    NodeModel *newNode = [self compareWithNode:node1 secondNode:node4];
    NSLog(@"newNode--->%@", newNode);
}

- (NodeModel *)compareWithNode:(NodeModel *)firstNode secondNode:(NodeModel *)secondNode {
    // 拦截判断
    if (!firstNode && secondNode) {
        return secondNode;
    } else if (firstNode && !secondNode) {
        return firstNode;
    } else if (!firstNode && !secondNode) {
        return nil;
    }
    NodeModel *minNode;
    // 递归，取出较大值与较小值的child继续比较
    if (firstNode.value > secondNode.value) {
        secondNode.next = [self compareWithNode:firstNode secondNode:secondNode.next];
        minNode = secondNode;
    } else {
        firstNode.next = [self compareWithNode:firstNode.next secondNode:secondNode];
        minNode = firstNode;
    }
    return minNode;
}

# 检测链表中是否有环

- (void)checkLinkedListHasCircle {
    NodeModel *node = [self createLinkedList];
    BOOL hasCircle = [self hasCircle:node];
    NSLog(@"链表中是否有环--->%@", hasCircle ? @"YES" : @"NO");
    
    // 如果要计算环的深度，记录下两个步进对象相遇的节点meetNode，然后stepOneNode继续往前走，直到遇见meetNode，遍历的次数就是链表的深度
}

- (BOOL)hasCircle:(NodeModel *)node {
    NodeModel *stepOneNode = node, *stepTwoNode = node;
    while (true) {
        stepOneNode = stepOneNode.next;
        stepTwoNode = stepTwoNode.next.next;
        if (!stepOneNode || !stepTwoNode) {
            return NO;
        }
        if (stepOneNode.value == stepTwoNode.value) {
            return YES;
        }
    }
    return NO;
}

# 链表翻转

- (void)linkedListReverse {
    NodeModel *headNode = [self createLinkedList];
    NodeModel *newNode = nil;
    NodeModel *cursorNode = headNode;
    while (cursorNode) {
        NodeModel *temp = cursorNode.next;// 保存child节点
        cursorNode.next = newNode;// child的值指向上一个保存的节点
        newNode = cursorNode;// 指针修改后将当前node赋值给newNode
        
        cursorNode = temp;// 右移继续遍历
    }
    NSLog(@"newNode--->%@", newNode);
}

# 两两交换链表中相邻的节点

/**
 给定一个链表，两两交换其中相邻的节点，并返回交换后的链表。
 示例:
 给定 1->2->3->4, 你应该返回 2->1->4->3.

 说明:
 你的算法只能使用常数的额外空间。
 你不能只是单纯的改变节点内部的值 ，而是需要实际的进行节点交换。
 */
- (void)swapNodesInPairsOfLinkedList {
    // 123579
    NodeModel *headNode = [self createLinkedList];
    
    NodeModel *newNode;// 新节点，第一次遍历时赋值就行了
    NodeModel *previousNode;
    NodeModel *firstNode = headNode;
    while (firstNode.next) {
        // 两两交换指针，第1个节点指针指向第3个，第2个指针指向第1个
        NodeModel *secondNode = firstNode.next;
        NodeModel *thirdNode = firstNode.next.next;
        firstNode.next = thirdNode;
        secondNode.next = firstNode;
        if (!newNode) {// 第一次循环后赋值即可
            newNode = secondNode;
        }
        if (previousNode) {
            previousNode.next = secondNode;// 上次循环记录的节点指针指向该组交换后的第一个节点
        }
        previousNode = firstNode;// 记录该组交换后的后一个节点
        firstNode = thirdNode;// 间隔跳，交换下一组指针
    }
    NSLog(@"newNode--->%@", newNode);
}

# 高效删除链表的倒数第N个节点

/**
 给定一个链表，删除链表的倒数第 n 个节点，并且返回链表的头结点。

 示例：
 给定一个链表: 1->2->3->4->5, 和 n = 2.
 当删除了倒数第二个节点后，链表变为 1->2->3->5.

 说明：
 给定的 n 保证是有效的，使用一趟扫描实现
 */
- (void)removeNthNodeFromEndOfLinkedList {
    // 这个问题的本质，倒数第n个就等正数的第（len-n）+1个
    // 利用双指针变量，前指针遍历到n时，后指针开始遍历，则for循环结束时，后指针刚好指到要删除的节点的前驱节点
    
    // 123579
    NodeModel *headNode = [self createLinkedList];
    int n = 2;
    
    int cursorIndex = 0;
    NodeModel *firstModel  = headNode;
    NodeModel *secondModel = headNode;
    while (firstModel.next) {
        cursorIndex ++;
        firstModel = firstModel.next;
        if (cursorIndex > n) {// 前指针大于n时，开始遍历后指针
            secondModel = secondModel.next;
        }
    }
    // 遍历结束时拿到的secondModel即为要删除的节点的前驱节点。
    NodeModel *tempNode = secondModel.next.next;
    secondModel.next = tempNode;
    NSLog(@"headNode--->%@", headNode);
}

# K个一组翻转链表

/**
 给你一个链表，每 k 个节点一组进行翻转，请你返回翻转后的链表。
 k 是一个正整数，它的值小于或等于链表的长度。
 如果节点总数不是 k 的整数倍，那么请将最后剩余的节点保持原有顺序。
 
 示例：
 给定一个链表: 1->2->3->5->7->9, 和 n = 3.
 输出 3->2->1->9->7->5.

 进阶：
 你可以设计一个只使用常数额外空间的算法来解决此问题吗？
 你不能只是单纯的改变节点内部的值，而是需要实际进行节点交换。
 */
- (void)reverseKGroup {
    int k = 3;
    // 123579
    NodeModel *headNode = [self createLinkedList];
    if (!headNode || k == 1) {
        return;
    }
    NodeModel *dummyNode = [NodeModel initWithValue:-1];
    NodeModel *preNode = dummyNode, *cursorNode = headNode;
    dummyNode.next = headNode;
    int index = 0;
    while (cursorNode) {
        index ++;
        if (index % k == 0) {
            preNode = [self reverseOneGroup:preNode lastNodeOfGroup:cursorNode.next];
            cursorNode = preNode.next;
        } else {
            cursorNode = cursorNode.next;
        }
    }
    NSLog(@"newNode--->%@", dummyNode.next);
}

/**
 翻转一组链表，可通过画图进行更好的理解
 */
- (NodeModel *)reverseOneGroup:(NodeModel *)preNode lastNodeOfGroup:(NodeModel *)lastNode {
    NodeModel *firstNode = preNode.next;
    NodeModel *cursorNode = firstNode.next;
    while (cursorNode != lastNode) {
        firstNode.next = cursorNode.next;// 第1个节点指向第3个节点，最终结果就是该组第1个节点(已翻转)指向下一组的第一个节点(未翻转)
        cursorNode.next = preNode.next;// 第2个节点指向第1个节点，翻转
        preNode.next = cursorNode;// 上一组的最后一个节点指向翻转后的第一个节点
        cursorNode = firstNode.next;// 游标右移
    }
    return firstNode;
}

# 两个链表的相交的起始节点

/**
 给你两个单链表的头节点 headA 和 headB ，请你找出并返回两个单链表相交的起始节点。如果两个链表不存在相交节点，返回 null 。
 题目数据 保证 整个链式结构中不存在环。
 注意，函数返回结果后，链表必须 保持其原始结构 。
 */
- (void)startNodeOfIntersectionOfTwoLinkedList {
    // 123579
    NodeModel *intersectionNode = [self createLinkedList];
    
    NodeModel *nodeA1 = [NodeModel initWithValue:10];
    NodeModel *nodeA2 = [NodeModel initWithValue:2];
    NodeModel *nodeB1 = [NodeModel initWithValue:3];
    nodeA1.next = nodeA2;
    nodeA2.next = intersectionNode;
    nodeB1.next = intersectionNode;
    NodeModel *resultNode = [self getIntersectionNode:nodeA1 secondNode:nodeB1];
    NSLog(@"resultNode--->%@", resultNode);
}

/**
 双指针解法，参考https://leetcode-cn.com/problems/intersection-of-two-linked-lists/solution/xiang-jiao-lian-biao-by-leetcode-solutio-a8jn/
 */
- (NodeModel *)getIntersectionNode:(NodeModel *)headA secondNode:(NodeModel *)headB {
    if (!headA || !headB) {
        return nil;
    }
    NodeModel *pA = headA, *pB = headB;
    while (pA != pB) {// 建立在相交节点开始，节点都是同一个地址
        pA = pA ? pA.next : headB;
        pB = pB ? pB.next : headA;
    }
    return pA;
}

# 删除排序链表中的重复元素

/**
 给定一个已排序的链表的头 head ， 删除原始链表中所有重复数字的节点，只留下不同的数字 。返回 已排序的链表 。
 
 示例 1：
 输入：head = [1,2,3,3,4,4,5]
 输出：[1,2,5]
 */
- (void)deleteDuplicates {
    NodeModel *headNode = [self createLinkedList];
    
    NodeModel *dummyNode = [NodeModel initWithValue:0];
    dummyNode.next = headNode;
    
    NodeModel *cur = dummyNode;
    while (cur.next && cur.next.next) {
        if (cur.next.value == cur.next.next.value) {
            int x = (int)cur.next.value;
            while (cur.next && cur.next.value == x) {
                cur.next = cur.next.next;
            }
        } else {
            cur = cur.next;
        }
    }
    
    NSLog(@"newNode--->%@", dummyNode.next);
}
