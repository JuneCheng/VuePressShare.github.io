# 找出不含有重复字符的最长子串的长度

/**
 给定 "abcabcbb" ，没有重复字符的最长子串是 "abc" ，那么长度就是3。
 给定 "bbbbb" ，最长的子串就是 "b" ，长度是1。
 给定 "pwwkew" ，最长子串是 "wke" ，长度是3。请注意答案必须是一个 子串 ， "pwke" 是 子序列 而不是子串。
 */
- (void)findLongestNotRepeatSubStirngLengthOfString {
    NSString *string = @"pwwkew";
    int maxLength = 0;
    for (int i = 0; i < string.length - 1; i ++) {
        for (int j = i; j < string.length - 1; j ++) {// j = i , 取下一个字符
            char nextCh = [string characterAtIndex:j + 1];
            int length = j - i + 1;// 需要截取的长度，注意+1
            NSString *subString = [string substringWithRange:NSMakeRange(i, length)];
            if ([subString containsString:[NSString stringWithFormat:@"%c", nextCh]]) {
                if (length > maxLength) {
                    maxLength = length;
                }
                break;
            }
        }
    }
    NSLog(@"maxLength--->%d", maxLength);
}

# 给定一个字符串s，找到s中最长的回文子串。假设s的最大长度为1000

/**
 示例 1：
 输入: "babad"
 输出: "bab"
 注意: "aba"也是一个有效答案。

 示例 2：
 输入: "cbbd"
 输出: "bb"
 */
- (void)longestPalindromicSubString {
    NSString *string = @"babcadacbaffs";
    NSString *palindromeLongest = @"";
    for (int i = 0; i < string.length; i ++) {
        NSString *cursorChar = [NSString stringWithFormat:@"%c", [string characterAtIndex:i]];
        for (int j = 1; j < 1000; j ++) {
            // 如果索引越界，中断
            if (i - j < 0 || i + j > string.length - 1) {
                break;
            }
            NSString *leftChar = [NSString stringWithFormat:@"%c", [string characterAtIndex:i - j]];
            NSString *rightChar = [NSString stringWithFormat:@"%c", [string characterAtIndex:i + j]];
            // 判断游标字符是否等于左右两侧字符，确定最短的回文
            if ([leftChar isEqualToString:cursorChar] || [rightChar isEqualToString:cursorChar]) {
                if (palindromeLongest.length < 2) {
                    palindromeLongest = [NSString stringWithFormat:@"%@%@", cursorChar, cursorChar];
                }
            }
            if ([leftChar isEqualToString:rightChar]) {
                int length = (i + j) - (i - j) + 1;// 左右字符对应，计算左右两字符相隔的距离
                if (palindromeLongest.length < length) {
                    palindromeLongest = [string substringWithRange:NSMakeRange(i - j, length)];
                }
            } else {
                break;
            }
        }
    }
    NSLog(@"longestPalindromicSubString--->%@", palindromeLongest);
}

# 将字符串 ABCDEFGHIJKLMN 以Z字形排列成给定的行数

/**
 将字符串 "ABCDEFGHIJKLMN" 以Z字形排列成给定的行数：

 A     E      I      M
 B D F H  J  L  N
 C    G     K
 之后从左往右，逐行读取字符： "AEIMBDFHJLNCGK"

 实现一个将字符串进行指定行数变换的函数:
 string convert(string s, int numRows);

 示例 1:
 输入: s = "ABCDEFGHIJKLMN", numRows = 3
 输出: "AEIMBDFHJLNCGK"

 示例 2:
 输入: s = "ABCDEFGHIJKLMN", numRows = 4
 输出: "AGMBFHLNCEIKDJ"
 解释:
 A           G             M
 B      F   H       L    N
 C  E       I    K
 D           J
 
 */
- (void)zigZagConversion {
    NSString *string = @"ABCDEFGHIJKLMN";
    int numRows = 3;
    NSString *convertString = [self zigZagConversion:string numRows:numRows];
    NSLog(@"convertString--->%@", convertString);
}

- (NSString *)zigZagConversion:(NSString *)string numRows:(int)numRows {
    if (numRows == 1) {
        return string;
    }
    // 创建空二位数组
    NSMutableArray<NSMutableArray *> *containArray = [NSMutableArray arrayWithCapacity:numRows];
    for (int i = 0; i < numRows; i ++) {
        containArray[i] = [NSMutableArray array];
    }
    int cycle = numRows * 2 - 2;// 关键点1：找出一个轮回的周期
    for (int i = 0; i < string.length; i ++) {
        NSString *chString = [NSString stringWithFormat:@"%c", [string characterAtIndex:i]];
        int indexInCycle = i % cycle;
        int remain = indexInCycle / numRows;// 关键点2：一个周期内 索引/行数 计算出正序还是倒序。余数为0说明不满numRows是正序，反之为倒序
        if (remain == 0) {
            [containArray[indexInCycle] addObject:chString];
        } else {
            // 关键点3：推导出 -2
            [containArray[numRows - indexInCycle % numRows - 2] addObject:chString];
        }
    }
    NSMutableString *mString = [NSMutableString string];
    for (NSMutableArray *array in containArray) {
        for (NSString *string in array) {
            [mString appendString:string];
        }
    }
    return [NSString stringWithFormat:@"%@", mString];
}

# 整数转罗马数字

/**
 罗马数字包含以下七种字符： I ， V ， X ， L ， C ， D 和 M 。
 字符 数值
 I 1
 V 5
 X 10
 L 50
 C 100
 D 500
 M 1000

 例如， 罗马数字 2 写做 II ，即为两个并列的 1。12 写做 XII ，即为 X + II 。 27 写做 XXVII , 即为 XX + V + II 。
 通常情况下，罗马数字中小的数字在大的数字的右边。但也存在特例，例如 4 不写做 IIII ，而是 IV 。数字 1 在数字 5 的左边，所表示的数等于大数 5 减小数 1 得到的数值 4 。同样地，数字 9 表示为 IX 。这个特殊的规则只适用于以下六种情况：

 I 可以放在 V (5) 和 X (10) 的左边，来表示 4 和 9。
 X 可以放在 L (50) 和 C (100) 的左边，来表示 40 和 90。
 C 可以放在 D (500) 和 M (1000) 的左边，来表示 400 和 900。
 给定一个整数，将其转为罗马数字。输入确保在 1 到 3999 的范围内。

 示例 1:
 输入: 3
 输出: "III"

 示例 2:
 输入: 4
 输出: "IV"

 示例 3:
 输入: 58
 输出: "LVIII"
 解释: C = 100, L = 50, XXX = 30, III = 3.

 示例 4:
 输入: 1994
 输出: "MCMXCIV"
 解释: M = 1000, CM = 900, XC = 90, IV = 4.
 */
- (void)convertNum2RomanNum {
    int number = 1994;
    
    NSArray *numArray = @[@(1000), @(900), @(500), @(400), @(100),@(90), @(50), @(40), @(10), @(9), @(5), @(4), @(1)];
    NSArray *romanArray = @[@"M", @"CM", @"D", @"CD", @"C", @"XC", @"L", @"XL", @"X", @"IX", @"V", @"IV", @"I"];
    
    NSMutableString *mString = [NSMutableString string];
    while (number > 0) {
        for (int i = 0; i < numArray.count; i ++) {
            int num = [numArray[i] intValue];
            if (number / num > 0) {// 大于0，加入对应的字母
                [mString appendString:romanArray[i]];
                number -= num;
                break;// 重新开始遍历
            } else {// 小于0，继续遍历
                continue;
            }
        }
    }
    NSLog(@"RomanNum--->%@", [NSString stringWithFormat:@"%@", mString]);
}

# 电话号码的字母组合

/**
 给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。
 给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。
 示例:
 输入："23"
 输出：["ad", "ae", "af", "bd", "be", "bf", "cd", "ce", "cf"].

 说明:
 尽管上面的答案是按字典序排列的，但是你可以任意选择答案输出的顺序
 */
- (void)letterCombinationsOfAPhoneNumber {
    NSString *inputString = @"23";
    
    NSArray *letterArray = @[@"", @"abc", @"def", @"ghi", @"jkl", @"mno", @"pqrs", @"tuv", @"wxyz"];
    NSMutableArray *targetArray = [NSMutableArray array];
    for (int i = 0; i < inputString.length; i ++) {
        int num = [[NSString stringWithFormat:@"%c", [inputString characterAtIndex:i]] intValue];
        [targetArray addObject:letterArray[num - 1]];
    }
    NSArray *resultArray = [self targetArray:targetArray resultArray:@[] startIndex:0];
    NSLog(@"resultArray--->%@", resultArray);
}

- (NSArray *)targetArray:(NSMutableArray *)targetArray resultArray:(NSArray *)resultArray startIndex:(NSInteger)startIndex {
    // 拦截，防止越界
    if (startIndex > targetArray.count - 1) {
        return resultArray;
    }
    NSString *letters = targetArray[startIndex];
    NSMutableArray *array = [NSMutableArray array];
    for (int i = 0; i < letters.length; i ++) {
        NSString *letter = [NSString stringWithFormat:@"%c", [letters characterAtIndex:i]];
        if (resultArray.count > 0) {// 如果数组已经有值了，说明至少是第二组字符串了，遍历之前的数组生成新的数组
            for (int j = 0; j < resultArray.count; j ++) {
                [array addObject:[NSString stringWithFormat:@"%@%@", resultArray[j], letter]];
            }
        } else {// 如果数组为空，说明是第一个字符串组遍历，将对应字母加入数组即可
            [array addObject:letter];
        }
    }
    return [self targetArray:targetArray resultArray:array startIndex:startIndex + 1];// 递归
}
